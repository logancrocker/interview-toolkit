Queues


A queue is a linear data structure that abides by the policy of First in, First out (FIFO). Queues are pretty straightforward. You can think of it as waiting in line for a ride at Disneyland, or as the British would say, "queueing".

Implementation


A queue can be implemented with anything like an array or linked list, just like stacks. The only difference is you add items to one end and remove them from the opposite end.
A Queue class must support four operations:
  1. add(item)
    Add an item to the end of the queue.
  2. remove()
    Remove the first (oldest) item in the queue.
  3. peek()
    Return the first (oldest) item in the queue.
  4. isEmpty()
    Return true if the queue is empty.

Examples



// C++
#include <iostream>

using namespace std;
    
class Queue {
  private:
    int queue[10];
    int size = 0;
    int first = 0;
    int last = 0;
    
  public:
    void add(int item) {
      if (last == 10) {
        cout << "Full queue!" << endl;
      }
      else {
        this->queue[last] = item;
        this->last += 1;
        this->size += 1;
      }
    }
    
    void remove() {
      if (first == last) {
        cout << "Empty queue!" << endl;
      }
      else {
        for (int i = 0; i < last - 1; i++) {
          this->queue[i] = queue[i + 1];
        }
        this->last -= 1;
        this->size -= 1;
      }
    }
    
    bool isEmpty() {
      return this->size == 0;
    }
    
    int peek() {
      if (this->size == 0) {
        cout << "Empty queue!" << endl;
        return -1;
      }
      else {
        return this->queue[first];
      }
    }
};
            

# Python
class Queue:

  def __init__(self):
    self.queue = []

  def isEmpty(self):
    return len(self.queue) == 0

  def add(self, data):
    self.queue.append(data)

  def remove(self):
    if (not len(self.queue) == 0):
      self.queue.pop(0)
    else:
      print("Empty queue!")
  
  def peek(self):
    if (not len(self.queue) == 0):
      return self.queue[0]
    else:
      print("Empty queue!")
